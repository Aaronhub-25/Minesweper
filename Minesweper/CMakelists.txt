cmake_minimum_required(VERSION 3.22)

project(Minesweper VERSION 0.1.0 LANGUAGES CXX)

file(GLOB_RECURSE SOURCES "src/*.cpp")

add_executable(Minesweper
    ${SOURCES}
)

set_property(TARGET Minesweper PROPERTY CXX_STANDARD 17)
 
# Link ncurses library
find_library(NCURSES_LIB ncurses)
if(NCURSES_LIB)
    target_link_libraries(Minesweper ${NCURSES_LIB})
else()
    # Fallback: try to link directly (works on most systems)
    target_link_libraries(Minesweper ncurses)
endif()
